lfs setstripe --size 3m -c 16 integer.dat 
aprun -n 160 -N 16 ./striping-benchmark.titan.exe -s 50GB -c integer > striping-titan.$PBS_JOBID.txt
source $MODULESHOME/init/bash
module load python_matplotlib
python graph.py -f striping-titan.$PBS_JOBID.txt --numNodes $PBS_NUM_NODES
